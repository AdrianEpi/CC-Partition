212 bool Partition::isSubsetSum(std::vector<int> v, int n, int sum) {
213  // Base Cases
214  if (sum == 0) {
215  return true;
216  }
217 
218  if (n == 0 && sum != 0) {
219  return false;
220  }
221 
222  // If last element is greater than sum, then ignore it
223  if (v[n - 1] > sum) {
224  return isSubsetSum(v, n - 1, sum);
225  }
226 
227  /*
228  else, check if sum can be obtained by any of the following
229  (a) including the last element
230  (b) excluding the last element
231  */
232  return isSubsetSum(v, n - 1, sum) || isSubsetSum(v, n - 1, sum - v[n - 1]);
233 }

256 bool Partition::dynamicFindPartition(void) {
257  sum_ = calculateSum(original_);
258 
259  // If sum is odd there can't be solution
260  if (sum_ % 2 != 0) {
261  return false;
262  }
263 
264  bool table[sum_ / 2 + 1][original_.size() + 1];
265 
266  // Initialize top row as true
267  for (unsigned i = 0; i <= original_.size(); i++) {
268  table[0][i] = true;
269  }
270 
271  // Initialize leftmost column, except table[0][0], as 0
272  for (int i = 1; i <= sum_ / 2; i++) {
273  table[i][0] = false;
274  }
275 
276  // Fill the partition table in bottom up manner
277  for (int i = 1; i <= sum_ / 2; i++) {
278  for (unsigned j = 1; j <= original_.size(); j++) {
279  table[i][j] = table[i][j - 1];
280  if (i >= original_[j - 1])
281  table[i][j] = table[i][j] || table[i - original_[j - 1]][j - 1];
282  }
283  }
284

287  for (int i = 0; i <= sum_ / 2; i++) {
288  for (unsigned j = 0; j <= original_.size(); j++) {
289  std::cout << table[i][j];
290  }
291  std::cout << std::endl;
292  }
293  */
294  return table[sum_ / 2][original_.size()];

319 bool Partition::findSolution (std::vector<int> v1, std::vector<int> v2, unsigned n) {
320  if (n >= original_.size()) {
321  if (calculateSum(v1) == calculateSum(v2)) {
322  set_V1(v1);
323  set_V2(v2);
324  return true;
325  }
326  return false;
327  }
328  else {
329  if (calculateSum(v1) > sum_ || calculateSum(v2) > sum_) {
330  return false;
331  }
332  std::vector<int> vv1 = v1;
333  vv1.push_back(original_[n]);
334  std::vector<int> vv2 = v2;
335  vv2.push_back(original_[n]);
336  return findSolution(vv1, v2, n + 1) || findSolution(v1, vv2, n + 1);
